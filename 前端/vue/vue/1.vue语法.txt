 		  指令语法





说明: 用在标签内部




v-text="属性"           	   				修改文本

v-bind:属性名="值"    	   				        追加属性
      :属性名="值"                      			简化版

v-html="属性"		  					修改标签
  
v-shou=" true 或 flase "          				条件显示(消耗低)

v-for="循环变量 in 数据结构"		                        循环显示
{{循环变量}}  	

v-if="true 或 flase "             				同上

v-on:事件="函数"      					        绑定事件

@事件="函数"							同上

v-model="属性"                             			双向绑定属性

v-指令:属性                                                     指令绑定属性

v-once								静态初始值

v-pre								vue功能失效
